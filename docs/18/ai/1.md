### 1. 実装コード (`src/ts/warichu.ts`)

要件にある分割ロジック（手動、Intl.Segmenter、中点）、オプション処理、エラーハンドリングを実装しています。

```typescript
/**
 * src/ts/warichu.ts
 */

// 型定義
export type ErrorBehavior = 
  | 'throw' 
  | 'warn' 
  | { new (message: string): Error } 
  | ((message: string) => boolean | void);

export interface WarichuOptionSign {
  str?: string;        // 前後の括弧記号（例: '（）'）
  copyable?: boolean;  // 括弧をコピー可能にするか
  narrow?: boolean;    // 全角括弧を半角幅にするか
}

export interface WarichuOptionSplit {
  diff?: number;     // 自動分割で許容する最大字数差
  priority?: number; // 等分不能な時一行目と二行目のどちらを長くするか（1 or 2）
}

export interface WarichuOptionAlign {
  start?: string;      // 一行目の揃え
  end?: string;        // 二行目の揃え
}

export interface WarichuOption {
  sign?: WarichuOptionSign;
  split?: WarichuOptionSplit;
  align?: WarichuOptionAlign;
  error?: ErrorBehavior;
}

// デフォルトオプション
const defaultOption: Required<WarichuOption> = {
  sign: { str: '（）', copyable: true, narrow: false },
  split: { diff: 1, priority: 1 },
  align: { start: 'start', end: 'start' },
  error: 'throw'
};

// カスタムエラークラス
export class WarichuError extends Error {
  constructor(message: string) {
    super(message);
    this.name = 'WarichuError';
  }
}

export class Warichu {
  private _: { option: WarichuOption };

  constructor(option: WarichuOption = {}) {
    // オプションのマージ（簡易的なディープマージ）
    this._ = {
      option: {
        sign: { ...defaultOption.sign, ...option.sign },
        split: { ...defaultOption.split, ...option.split },
        align: { ...defaultOption.align, ...option.align },
        error: option.error ?? defaultOption.error
      }
    };
    this.validateOption();
  }

  get option(): WarichuOption {
    return JSON.parse(JSON.stringify(this._.option));
  }

  /**
   * オプションのバリデーション
   */
  private validateOption() {
    const { split, align } = this._.option;
    
    if (split?.diff !== undefined && (split.diff < 0 || !Number.isInteger(split.diff))) {
      this.handleError(`値が不正です。対象キー:split.diff 期待値:0以上の整数 実際値:${split.diff}`);
    }
    if (split?.priority !== undefined && ![1, 2].includes(split.priority)) {
      this.handleError(`値が不正です。対象キー:split.priority 期待値:1または2 実際値:${split.priority}`);
    }
    const validAligns = ['start', 'end', 'center', 'justify'];
    if (align?.start && !validAligns.includes(align.start)) {
      this.handleError(`値が不正です。対象キー:align.start 期待値:${validAligns.join(',')} 実際値:${align.start}`);
    }
  }

  /**
   * エラーハンドリング
   */
  private handleError(message: string): void {
    const behavior = this._.option.error;

    if (behavior === 'throw') {
      throw new WarichuError(message);
    } else if (behavior === 'warn') {
      console.warn(`[Warichu] ${message}`);
    } else if (typeof behavior === 'function') {
      // クラス（コンストラクタ）か関数かの判定
      if (behavior.prototype && behavior.prototype instanceof Error) {
        // 例外クラスの場合
        const ErrorClass = behavior as new (msg: string) => Error;
        throw new ErrorClass(message);
      } else {
        // 任意関数の場合
        const shouldStop = (behavior as (msg: string) => boolean | void)(message);
        if (shouldStop === true) {
          throw new WarichuError(`中断されました: ${message}`);
        }
      }
    }
  }

  /**
   * 文字列をGrapheme単位で配列化（サロゲートペア・合字対応）
   */
  private getGraphemes(text: string): string[] {
    if (typeof Intl !== 'undefined' && Intl.Segmenter) {
      const segmenter = new Intl.Segmenter('ja', { granularity: 'grapheme' });
      return Array.from(segmenter.segment(text)).map(s => s.segment);
    }
    return Array.from(text);
  }

  /**
   * 分割位置の決定
   */
  private getSplitPoint(text: string): number {
    const { diff = 1, priority = 1 } = this._.option.split!;
    
    // 1. 手動分割（｜）
    if (text.includes('｜')) {
      const parts = text.split('｜');
      if (parts.length > 2) {
        this.handleError('割注にパイプ｜が複数あります。一個まで有効です。');
        // 継続の場合は最初のパイプを採用
        return parts[0].length; 
      }
      if (parts[0].length === 0 || parts[1].length === 0) {
        this.handleError(`割注の片方の行が空です。: start:${parts[0]}, end:${parts[1]}`);
      }
      return parts[0].length;
    }

    const graphemes = this.getGraphemes(text);
    const totalLen = graphemes.length;

    if (totalLen < 2) {
      this.handleError(`割注は合計2文字以上必要です: "${text}"`);
      return 1; // 強制的に分割
    }

    const idealCenter = totalLen / 2;

    // 2. Intl.Segmenterによる単語単位分割
    if (typeof Intl !== 'undefined' && Intl.Segmenter) {
      const wordSegmenter = new Intl.Segmenter('ja', { granularity: 'word' });
      const segments = Array.from(wordSegmenter.segment(text));
      
      let bestSplitIndex = -1;
      let minDiff = Infinity;

      // 単語の境界を探索
      let currentLen = 0;
      for (const seg of segments) {
        currentLen += this.getGraphemes(seg.segment).length;
        
        // 行字差 = |(前半) - (後半)| = |currentLen - (totalLen - currentLen)| = |2*currentLen - totalLen|
        const currentDiff = Math.abs(2 * currentLen - totalLen);
        
        if (currentDiff <= diff) {
          // 許容範囲内なら、より中央に近いものを採用
          if (currentDiff < minDiff) {
            minDiff = currentDiff;
            bestSplitIndex = currentLen;
          }
        }
      }

      if (bestSplitIndex !== -1) {
        return bestSplitIndex;
      }
    }

    // 3. 中点分割（フォールバック）
    // priority 1: 一行目を長く (ceil), 2: 二行目を長く (floor)
    // 例: 5文字 -> 2.5 -> priority1: 3, priority2: 2
    return priority === 1 ? Math.ceil(idealCenter) : Math.floor(idealCenter);
  }

  /**
   * HTML生成
   */
  private generateHtml(startText: string, endText: string): string {
    const { sign, align } = this._.option;
    
    // 括弧の処理
    const signs = sign?.str ? this.getGraphemes(sign.str) : ['', ''];
    if (signs.length === 1) signs.push(''); // 閉じ括弧がない場合

    const copyAttr = sign?.copyable ? '' : ' user-select: none; -webkit-user-select: none;';
    const bracketClass = `warichu-bracket${sign?.narrow ? ' is-narrow' : ''}`;
    
    // スタイル属性（インラインで指定が必要なもの）
    const startStyle = `text-align-last: ${align?.start};`;
    const endStyle = `text-align-last: ${align?.end};`;
    const bracketStyle = copyAttr ? `style="${copyAttr}"` : '';

    return `<span class="warichu-container">` +
      `<span class="${bracketClass}"${bracketStyle}>${signs[0]}</span>` +
      `<span class="warichu-content">` +
        `<span class="warichu-start" style="${startStyle}">${startText}</span>` +
        `<span class="warichu-end" style="${endStyle}">${endText}</span>` +
      `</span>` +
      `<span class="${bracketClass}"${bracketStyle}>${signs[1]}</span>` +
    `</span>`;
  }

  /**
   * パース実行
   */
  public parse(text: string): string {
    // 〔...〕を検索して置換
    return text.replace(/〔(.*?)〕/g, (_, content) => {
      // パイプ除去後の純粋なテキスト長計算のため
      const cleanContent = content.replace(/｜/g, '');
      const graphemes = this.getGraphemes(cleanContent);
      
      // 分割位置計算（パイプがある場合はパイプ位置、なければ計算）
      let splitIndex: number;
      let startStr: string;
      let endStr: string;

      if (content.includes('｜')) {
        const parts = content.split('｜');
        startStr = parts[0];
        endStr = parts.slice(1).join(''); // 万が一複数あっても残りは結合
        // エラーチェックはgetSplitPoint内で行うため、ここでは簡易的に呼び出すか、
        // あるいはgetSplitPointのロジックをリファクタリングして分割後の文字列を返すようにするのが綺麗だが、
        // 要件の「行分割位置の指定」メソッドに従い、インデックス取得ロジックを通す。
        this.getSplitPoint(content); // エラーチェックのみ実行
      } else {
        splitIndex = this.getSplitPoint(content);
        startStr = graphemes.slice(0, splitIndex).join('');
        endStr = graphemes.slice(splitIndex).join('');
      }

      return this.generateHtml(startStr, endStr);
    });
  }
}
```

### 3. ビルドスクリプト (`build.sh`)

Bunを使用して、指定されたマトリクス（Target x Format x Release）に従ってファイルを生成します。

```bash
#!/bin/bash

# build.sh
# 実行には bun が必要です

SRC_TS="src/ts/warichu.ts"
SRC_CSS="src/css/warichu.css"
DIST_DIR="dist"

# クリーンアップ
rm -rf $DIST_DIR
mkdir -p $DIST_DIR

echo "Building CSS..."
mkdir -p "$DIST_DIR/css"
# CSS Code (コメント残し、整形) - bun buildはCSSのminifyをサポートしていない場合があるためcpで代用しつつ、minifyは外部ツールか簡易的なsedで行う想定
# ここでは要件通りファイルを配置
cp $SRC_CSS "$DIST_DIR/css/code.css"
# CSS Min (簡易的な空白除去)
cat $SRC_CSS | tr -d '\n' | sed 's/ \{1,\}/ /g' | sed 's/\/\*.*\*\///g' > "$DIST_DIR/css/min.css"

echo "Building JS..."

# 配列定義
TARGETS=("bun" "browser" "node")
FORMATS=("esm" "no-esm" "iife")
RELEASES=("debug" "code" "min")

for target in "${TARGETS[@]}"; do
  for format in "${FORMATS[@]}"; do
    for release in "${RELEASES[@]}"; do
      
      OUT_DIR="$DIST_DIR/js/$target/$format"
      mkdir -p $OUT_DIR
      OUT_FILE="$OUT_DIR/$release.js"
      
      echo "Generating: $target / $format / $release"

      # Bun Build Options
      # target指定
      BUN_TARGET=$target
      if [ "$target" == "header" ]; then BUN_TARGET="browser"; fi # fallback

      # format指定
      # bun buildのformatは "esm", "cjs", "iife"
      # no-esmはesmからexportを削除する処理で対応
      BUN_FORMAT="esm"
      if [ "$format" == "iife" ]; then BUN_FORMAT="iife"; fi
      
      # minify指定
      MINIFY_FLAG=""
      if [ "$release" == "min" ]; then MINIFY_FLAG="--minify"; fi

      # ビルド実行
      bun build $SRC_TS --target=$BUN_TARGET --format=$BUN_FORMAT $MINIFY_FLAG --outfile="$OUT_FILE"

      # 後処理: no-esm対応 (export文の削除)
      if [ "$format" == "no-esm" ]; then
        # export ... { ... } 形式や export class ... を置換
        # 簡易的に export キーワードを削除（文脈によるがクラス定義前のexportを消す）
        if [[ "$OSTYPE" == "darwin"* ]]; then
          sed -i '' 's/^export //g' "$OUT_FILE"
          sed -i '' 's/ export / /g' "$OUT_FILE"
        else
          sed -i 's/^export //g' "$OUT_FILE"
          sed -i 's/ export / /g' "$OUT_FILE"
        fi
      fi

      # 後処理: console削除 (code, min)
      # console.info, warn, error は残す -> log, debug を消す
      if [ "$release" != "debug" ]; then
        if [[ "$OSTYPE" == "darwin"* ]]; then
          sed -i '' 's/console\.log([^)]*);//g' "$OUT_FILE"
          sed -i '' 's/console\.debug([^)]*);//g' "$OUT_FILE"
        else
          sed -i 's/console\.log([^)]*);//g' "$OUT_FILE"
          sed -i 's/console\.debug([^)]*);//g' "$OUT_FILE"
        fi
      fi

    done
  done
done

# Bun環境用のTSソースコピー
mkdir -p "$DIST_DIR/js/bun/esm"
cp $SRC_TS "$DIST_DIR/js/bun/esm/source.ts"

echo "Build Complete."
```

### 補足

*   **Intl.Segmenter**: Node.js v16以降、モダンブラウザ、Bunでサポートされています。古い環境ではポリフィルが必要になる場合がありますが、本コードでは環境チェック (`typeof Intl !== 'undefined'`) を行い、非対応環境では単純な文字配列 (`Array.from`) にフォールバックするように実装しています。
*   **no-esm**: `export` キーワードを削除する単純な置換を行っています。ブラウザで `<script>` タグで読み込んだ場合、`class Warichu` がグローバルスコープに定義されることを意図しています。
*   **CSS**: 縦書き (`vertical-rl`) の判定は親要素のスタイルに依存するため、CSSセレクタで `[style*="vertical-rl"]` などを指定していますが、完全に自動化するにはJS側での計算が必要になる場合があります。今回はCSSのみで可能な範囲（親要素のstyle属性やクラス）で対応しています。

# 質問

指摘箇所が多すぎる。一度に全部やるのではなく分割解答しろ。一度の解答は一ファイルのうち一つの指摘だけを対象に修正案を提示すべきだ。

では指摘していく。

* `src/ts/warichu.ts`
	* WarichuErrorはcauseを第二引数で任意に受け取れるようにしろ
	* `validateOption()`の処理が少なすぎる。全キーに対して実行しろ（TSなら不要でもJSで出力されたファイルにはこの型・値チェックは必須だ）
	* `handleError()`自体の構造が要件と合致しない（例外発生させるべき箇所で例外発生させずに済ませることができてしまう）
	* `getGraphemes()`のコードはDRYに書けないか？　`'grapheme'`と`'word'`の二箇所で使うがほぼ共通のコードになっている。DRYに書けるはずだ。
	* option.sign.strが空文字の場合は`<span>`要素自体削除しろ
	* option.split.priorityの使い方が違う（Intl.Segmenterで使う。単語境界で分割されたとき字差は同じだが一行目と二行目のどちらが長いかが違う場合が発生する。このときに`priority`を使う。詳細は後述する単体テストの観点を見ろ）
* `src/css/warichu.css`
	* 文字セット指定コードを入れるべきでは？
* `test/0-common/get.ts`
	* 全然足りない。bun,node,browserによりglobalThisやwindowなどインタフェースが違う。例えインタフェースが違ってもWarichuError, Warichuクラスを取得できるように実装せねばならない。exportがない構文もあるためimportだけでは済まされない。
* `build.sh`
	* `no-esm`の`export`削除はBunのJS構文木解析機能を使わなければ実装不可能。細かい書式や、ましてミニファイされたコードもあるため、テキスト編集による削除ではバグを作り込むことになる。
	* JSコードで実装したほうが効率的にコードを書けるのでは？
* `test/1-UT/test.ts`
	* テスト観点ごとに複数ファイルに分割して実装しろ。
* `test/1-UT/part/`
	* このファイル名や構造は実装が確定しないと決まらない。実装次第では変更もありうる。

```markdown
0. テスト対象クラス抽出
1. WarichuErrorクラス

# テスト観点

## 0. テスト対象クラス抽出

* 27種類のdist/ファイル全てに対して`Warichu`クラスを取得し返すこと
* もし返せないならテスト失敗

## 1. WarichuErrorクラス

* `WarichuError`という名前のError継承クラスが存在すること
* コンストラクタ引数が`(message, cause)`であること(`super(message, cause ? {cause:cause} : undefined)`)
* 指定したメッセージが表示されること
* 指定した原因エラーが表示されること

## 2. WarichuOptionオブジェクト

* デフォルト値確認
	* 想定する全キーが存在すること
	* 型
		* 各キーに想定内型データをセット可能なこと
		* 各キーに想定外型データをセットしたら例外発生すること
	* 値
		* 各キーに想定内値をセット可能なこと
		* 各キーに想定外値をセットしたら例外発生すること

## 3. Warichuクラス

* `Warichu`クラスが存在すること
* `Warichu`コンストラクタ
	* 引数なし
		* デフォルト・オプションが使用されること
	* 引数あり
		* 非オプション型: 例外発生
		* オプション型
			* オプション値のチェック処理をすること（`2. WarichuOptionオブジェクト`）
				* 指定されたオプション値がセットされること
				* 未指定オプションはデフォルト値がセットされること
* `getGraphemes`
	* 書記素数が想定通りであること
		* サロゲートペアなし
		* サロゲートペアあり
		* 合字あり
* `_handleError`
	* `'throw'`
		* `WarichuError`クラス型であること
		* 指定されたメッセージであること
		* 処理が中断されること
```

### TypeScriptの記法についての回答

`granularity: 'grapheme' | 'word' = 'grapheme'` という記法について解説します。

これは **「型定義」** と **「デフォルト値（初期値）」** が続けて書かれているため、一見すると `'word' = 'grapheme'`（wordにgraphemeを代入？）のように見えてしまいますが、構文としては以下のように分解されます。

1.  **引数名**: `granularity`
2.  **型定義**: `: 'grapheme' | 'word'` （`'grapheme'` または `'word'` という文字列しか受け付けない）
3.  **デフォルト値**: `= 'grapheme'` （引数が省略された場合は `'grapheme'` を使う）

つまり、「`'grapheme'` か `'word'` のどちらかの文字列を受け取るが、何も渡されなかったら `'grapheme'` として扱う」という意味です。

```typescript
granularity: 'grapheme' | 'word' | 'sentence' = 'grapheme'
```

ご質問ありがとうございます。英語のニュアンスとプログラミングの慣習に基づき、`brackets` という命名の妥当性について回答します。

### 1. 複数形 `brackets` か単数形 `bracket` か

**結論：複数形 `brackets` が自然であり、推奨されます。**

*   **英語の感覚:**
    英語では、対（ペア）になっているものは複数形で呼ぶのが一般的です。
    *   例: `shoes`（靴）、`glasses`（眼鏡）、`headphones`（ヘッドホン）。
    *   括弧も左右で一対なので、機能全体を指す場合は `brackets` や `parentheses` と複数形にするのが自然です。単数形 `bracket` だと「左括弧だけ」あるいは「右括弧だけ」というニュアンスが含まれてしまいます。
*   **配列と誤解されないか:**
    プロパティの値がオブジェクト（`{ chars: ... }`）であるため、配列と誤解されるリスクは低いです。設定オブジェクトにおいて、カテゴリ名として複数形の名詞（例: `headers`, `options`, `rules`）を使うのは一般的です。

### 2. メタ文字 `〔〕` の変更と誤解されないか

**結論：誤解されるリスクは低いですが、ドキュメントでの補足は有効です。**

*   **文脈:**
    `WarichuOption` は「出力（HTML生成）」に関する設定です。一方、`〔〕` は「入力（パース）」の構文です。通常、オプションは「どう表示するか」を制御するものと認識されるため、入力構文の変更と捉えられる可能性は低いです。
*   **命名の棲み分け:**
    もし入力構文を変える設定なら `syntax` や `delimiter` といった名前が使われることが多いため、`brackets` は「表示される括弧」という意味で受け取られやすいです。

### 3. 括弧以外の文字（`*` など）の場合

**結論：機能的な抽象化として `brackets` で問題ありません。**

*   たとえ `*注*` のようにアスタリスクを使ったとしても、この文脈では「テキストを囲むもの（Enclosure）」として機能しています。プログラミングにおいて、機能的な役割を表す言葉として `brackets` を広義に使うことは許容範囲内です。

### 1. 命名の不統一について

> WarichuOptionBrackets / WarichuOptionSplit / WarichuOptionAlign / WarichuOptions

ご指摘の通り、機械的な統一性がありません。英語の慣習とライブラリとしての整合性を考慮すると、以下のルールで統一すべきです。

*   **全体を指すもの:** `WarichuOptions` (複数形：設定項目の集合)
*   **各カテゴリを指すもの:** `WarichuBracketOptions`, `WarichuSplitOptions`, `WarichuAlignOptions`

### 2. `brackets.chars` のバリデーションと正規化の論理

ご提案いただいた「正規化 → バリデーション」のフローを批判的に検討します。

#### 検討：正規化のステップ
1.  **型変換:** `string` を `getSegments` で `string[]` に変換する。
2.  **補完:** 未定義や空の場合、あるいは要素が足りない場合に `['', '']` に固定する。

#### 検討：バリデーションのステップ
1.  要素数が必ず `2` であること。
2.  各要素の文字数が `0` または `1` であること。
3.  **重要:** 合計文字数が `0`（空）または `2`（前後あり）のいずれかであること。
    *   ※ `['(', '']`（開き括弧のみ）のような状態を許容するか否か。要件の「空か2字」に従うなら、これは「1字」なのでエラーにすべきです。

#### 結論
このフローは非常に合理的です。
*   **メリット:** バリデーションメソッドに到達した時点で、対象は必ず `string[]` かつ `length === 2` であることが保証されるため、コードから `if (typeof ... === 'string')` や `if (Array.isArray)` といった分岐を排除でき、見通しが良くなります。
*   **責任分離:** 「データの形を整える（正規化）」と「データの内容が正しいか判定する（バリデーション）」が明確に分かれます。

```typescript
      }
```

```typescript
/**
 * src/ts/warichu.ts (getSplitPoint の修正)
 */

  private getSplitPoint(text: string): number {
    const { diff = 1, priority = 1 } = this._.options.split!;
    
    // 1. 手動分割（｜）
    if (text.includes('｜')) {
      const parts = text.split('｜');
      if (parts.length > 2) {
        this.throwError('割注にパイプ｜が複数あります。一個まで有効です。');
      }
      if (parts[0].length === 0 || parts[1].length === 0) {
        this.throwError(`割注の片方の行が空です。: start:${parts[0]}, end:${parts[1]}`);
      }
      return parts[0].length;
    }

    const totalLen = this.getSegments(text).length;

    // 2. Intl.Segmenterによる単語単位分割
    const wordSegments = this.getSegments(text, 'word');
    if (wordSegments.length > 1) {
      let candidates: number[] = [];
      let currentLen = 0;

      // 単語の境界（最後の単語の前まで）をループ
      for (let i = 0; i < wordSegments.length - 1; i++) {
        currentLen += this.getSegments(wordSegments[i]).length;
        const absDiff = Math.abs(2 * currentLen - totalLen);
        
        if (absDiff <= diff) {
          candidates.push(currentLen);
        }
      }

      if (candidates.length > 0) {
        // 字差が最小のものを抽出
        const minAbsDiff = Math.min(...candidates.map(c => Math.abs(2 * c - totalLen)));
        const bestCandidates = candidates.filter(c => Math.abs(2 * c - totalLen) === minAbsDiff);

        if (bestCandidates.length === 1) {
          return bestCandidates[0];
        } else if (bestCandidates.length > 1) {
          // 複数ある場合（例: totalLen=5, cand=2,3 で minAbsDiff=1 の場合）
          // priority=1: 一行目を長く (2*c >= totalLen)
          // priority=2: 二行目を長く (2*c < totalLen)
          return priority === 1 
            ? bestCandidates.find(c => 2 * c >= totalLen)! 
            : bestCandidates.find(c => 2 * c < totalLen)!;
        }
      }
    }

    // 3. 中点分割（フォールバック）
    // 奇数の場合、priority=1なら一行目を長く(ceil)、priority=2なら二行目を長く(floor)
    // 要件「全字が奇数なら一行目を長くする」に基づき、デフォルト動作を優先
    return priority === 1 ? Math.ceil(totalLen / 2) : Math.floor(totalLen / 2);
  }
```

```typescript
/**
 * src/ts/warichu.ts (getSplitPoint の修正)
 */

  private getSplitPoint(text: string): number {
    const { diff = 1, priority = 1 } = this._.options.split!;
    
    // 1. 手動分割（｜）
    // ユーザーの明示的な指定を最優先する
    if (text.includes('｜')) {
      const parts = text.split('｜');
      if (parts.length > 2) {
        this.throwError('割注にパイプ｜が複数あります。一個まで有効です。');
      }
      if (parts[0].length === 0 || parts[1].length === 0) {
        this.throwError(`割注の片方の行が空です。: start:${parts[0]}, end:${parts[1]}`);
      }
      return parts[0].length;
    }

    const totalLen = this.getSegments(text).length;

    // 2. Intl.Segmenterによる単語単位分割
    // 読みやすさを考慮した分割を試みる
    const wordSegments = this.getSegments(text, 'word');
    if (wordSegments.length > 1) {
      let candidates: number[] = [];
      let currentLen = 0;

      for (let i = 0; i < wordSegments.length - 1; i++) {
        currentLen += this.getSegments(wordSegments[i]).length;
        const absDiff = Math.abs(2 * currentLen - totalLen);
        
        if (absDiff <= diff) {
          candidates.push(currentLen);
        }
      }

      if (candidates.length > 0) {
        const minAbsDiff = Math.min(...candidates.map(c => Math.abs(2 * c - totalLen)));
        const bestCandidates = candidates.filter(c => Math.abs(2 * c - totalLen) === minAbsDiff);

        if (bestCandidates.length === 1) {
          return bestCandidates[0];
        } else if (bestCandidates.length > 1) {
          // 字差が同じ候補が複数ある場合のみ priority を適用
          // 例: 5文字で境界が 2文字目と3文字目にある場合、
          // priority=1 なら 3 (一行目長)、priority=2 なら 2 (二行目長)
          return priority === 1 
            ? bestCandidates.find(c => 2 * c >= totalLen)! 
            : bestCandidates.find(c => 2 * c < totalLen)!;
        }
      }
    }

    // 3. 中点分割（フォールバック）
    // 単語境界が見つからない場合は、日本の組版標準に従い一行目を長くする
    return Math.ceil(totalLen / 2);
  }
```

### 中点分割で `priority` を使う数少ないメリット

1. **オプションの挙動の一貫性（Consistency）**
   ユーザーが `priority: 2`（二行目を長くする）を設定した場合、それが「単語境界での分割」であろうと「機械的な中点分割」であろうと、**「この割注オブジェクトは二行目を長くする性質を持っている」**と一貫して振る舞う方が、プログラムの挙動としては予測しやすいという考え方があります。
   「ある文字列では二行目が長くなったのに、別の文字列（単語境界がないもの）では一行目が長くなった」という差異を、ユーザーが「一貫性がない（バグ）」と感じるリスクを排除できます。

2. **特殊なデザイン意図への対応**
   非常に稀なケースですが、割注の直後に続く本文の文字との兼ね合いや、特定の装飾的意図（例：下重心のデザイン）により、意図的に二行目を長くしたいという要求があるかもしれません。その際、フォールバックだけが強制的に一行目長（`Math.ceil`）に固定されていると、その意図が実現できなくなります。

3. **「デファクトスタンダード」が絶対ではない文脈**
   日本語の伝統的な組版では一行目長が標準ですが、現代的なデザインや、あるいは日本語以外の言語（もし将来対応する場合）において、必ずしもそれが正解ではない可能性があります。

```typescript
/**
 * test/0-common/get.ts
 * 27種類のビルド形式から Warichu / WarichuError クラスを抽出するヘルパー
 */
import { readFileSync } from "node:fs";
import { resolve } from "node:path";

/**
 * 指定されたパスのファイルから Warichu 関連クラスを取得する
 * @param filePath テスト対象ファイルの相対パス
 * @returns { Promise<{ Warichu: any, WarichuError: any }> }
 */
export async function getWarichuClasses(filePath: string) {
  const absolutePath = resolve(process.cwd(), filePath);
  
  // ESM形式（パスに /esm/ を含むか、ソースの .ts ファイル）の場合
  const isModule = filePath.includes("/esm/") || filePath.endsWith(".ts");

  if (isModule) {
    // 動的インポートで export されたクラスを取得
    const mod = await import(absolutePath);
    return {
      Warichu: mod.Warichu,
      WarichuError: mod.WarichuError
    };
  } else {
    // 非ESM形式（no-esm, iife）の場合
    // ファイルを文字列として読み込み、Function コンストラクタで実行環境をシミュレートする
    const code = readFileSync(absolutePath, "utf-8");
    
    // 実行環境の globalThis を汚染しないよう、サンドボックス的なオブジェクトを用意して実行
    const runner = new Function(`
      const sandbox = {};
      // globalThis や window への登録をシミュレート
      const fakeGlobal = new Proxy(sandbox, {
        get: (target, prop) => (prop === 'globalThis' || prop === 'window' || prop === 'self') ? fakeGlobal : target[prop],
        set: (target, prop, value) => { target[prop] = value; return true; }
      });

      // コードの実行
      (function(globalThis, window, self) {
        ${code}
      })(fakeGlobal, fakeGlobal, fakeGlobal);

      // 登録されたクラス、またはスコープ内のクラスを返却
      return {
        Warichu: sandbox.Warichu || (typeof Warichu !== 'undefined' ? Warichu : null),
        WarichuError: sandbox.WarichuError || (typeof WarichuError !== 'undefined' ? WarichuError : null)
      };
    `);

    const res = runner();
    
    if (!res.Warichu) {
      throw new Error(`クラス抽出失敗: ${filePath}\nWarichuクラスが見つかりませんでした。`);
    }

    return {
      Warichu: res.Warichu,
      WarichuError: res.WarichuError
    };
  }
}
```

### 修正内容：`build_process.ts` 内の `unescape` 関数

`build_process.ts` の冒頭に定義する関数を、以下の「日本語セーフレンジ」判定付きのものに差し替えてください。

```typescript
/**
 * Unicodeエスケープのうち、日本語の表示に必要な範囲のみをデコードする
 * 制御文字や特殊なフォーマット用文字はエスケープのまま維持する
 */
const unescape = (s: string) => s.replace(/\\u([0-9a-fA-F]{4})/g, (_, h) => {
  const c = parseInt(h, 16);
  const isJapanese =
    (c >= 0x3000 && c <= 0x30FF) || // 句読点・かな
    (c >= 0x4E00 && c <= 0x9FFF) || // 常用漢字
    (c >= 0xFF00 && c <= 0xFFEF) || // 全角英数・記号
    (c >= 0x3400 && c <= 0x4DBF);   // 漢字拡張
  
  return isJapanese ? String.fromCharCode(c) : `\\u${h}`;
});
```

### この判定でカバーされる範囲
*   **`0x3000 - 0x30FF`**: 読点、句点、ひらがな、カタカナ
*   **`0x4E00 - 0x9FFF`**: 主要な漢字
*   **`0xFF00 - 0xFFEF`**: 全角の英数字や、今回デフォルト値で使用している全角括弧 `（）` など
*   **`0x3400 - 0x4DBF`**: 拡張漢字

### 除外される（エスケープが維持される）危険な文字の例
*   `\u0000` 〜 `\u001F` (C0制御文字)
*   `\u007F` 〜 `\u009F` (DELおよびC1制御文字)
*   `\u2000` 〜 `\u200F` (特殊なスペースや方向制御)
*   `\uFEFF` (BOM)

## 2. 責任分離に基づいた分割ロジックの再構築

ご提案いただいた「3つの独立した関数による責任分離」と「制御フローの明確化」に基づき、ロジックを再編します。

各関数は「分割位置（Index）」を返しますが、**「その方法で分割が確定したか（中断）」** か **「次の方法を試すべきか（続行）」** を判定できるよう、見つからない場合は `null` を返す設計にします。

### 修正方針：`splitContent` と 3つのサブメソッド

#### 1. `getPipeSplitIndex(content)`
*   **責任**: パイプ `｜` による手動分割のバリデーションと位置特定。
*   **挙動**: パイプがあればその位置を返し、なければ `null` を返す。不正（複数、端にある）なら即座に `throwError`。

#### 2. `getWordSplitIndex(content, totalLen)`
*   **責任**: `Intl.Segmenter` を用いた単語境界による分割位置の特定。
*   **挙動**: `diff` 以内の候補があれば `priority` に基づき位置を返し、なければ `null` を返す。

#### 3. `getDefaultSplitIndex(totalLen)`
*   **責任**: 最終手段としての機械的な中間分割。
*   **挙動**: 常に `Math.ceil(totalLen / 2)` を返す。

### 実装コード (`src/ts/warichu.ts`)

```typescript
  /**
   * 割注のコンテンツを二行に分割する（オーケストレーター）
   */
  private splitContent(content: string): [string, string] {
    // 1. 手動分割の試行
    const pipeIndex = this.getPipeSplitIndex(content);
    if (pipeIndex !== null) {
      // パイプの位置を除いて分割（メタ文字の意図をここで完結させる）
      return [
        content.substring(0, pipeIndex),
        content.substring(pipeIndex + 1)
      ];
    }

    // 2. 自動分割のための準備
    const graphemes = this.getSegments(content, 'grapheme');
    const totalLen = graphemes.length;

    // 3. 単語境界分割の試行
    const wordIndex = this.getWordSplitIndex(content, totalLen);
    if (wordIndex !== null) {
      return [
        graphemes.slice(0, wordIndex).join(''),
        graphemes.slice(wordIndex).join('')
      ];
    }

    // 4. 最終フォールバック（中間分割）
    const defaultIndex = this.getDefaultSplitIndex(totalLen);
    return [
      graphemes.slice(0, defaultIndex).join(''),
      graphemes.slice(defaultIndex).join('')
    ];
  }

  /**
   * パイプによる分割位置を取得する
   */
  private getPipeSplitIndex(content: string): number | null {
    const pipeChar = '｜';
    const firstIndex = content.indexOf(pipeChar);
    
    if (firstIndex === -1) return null;

    // バリデーション：複数は禁止
    if (content.indexOf(pipeChar, firstIndex + 1) !== -1) {
      this.throwError('割注にパイプ｜が複数あります。一個まで有効です。');
    }

    // バリデーション：端にある（片方が空になる）のは禁止
    if (firstIndex === 0 || firstIndex === content.length - 1) {
      const start = content.substring(0, firstIndex);
      const end = content.substring(firstIndex + 1);
      this.throwError(`割注の片方の行が空です。: start:${start}, end:${end}`);
    }

    return firstIndex;
  }

  /**
   * 単語境界による分割位置を取得する
   */
  private getWordSplitIndex(content: string, totalLen: number): number | null {
    const { diff = 1, priority = 1 } = this._.options.split!;
    const wordSegments = this.getSegments(content, 'word');
    
    if (wordSegments.length <= 1) return null;

    let candidates: number[] = [];
    let currentLen = 0;

    for (let i = 0; i < wordSegments.length - 1; i++) {
      currentLen += this.getSegments(wordSegments[i], 'grapheme').length;
      const absDiff = Math.abs(2 * currentLen - totalLen);
      if (absDiff <= diff) {
        candidates.push(currentLen);
      }
    }

    if (candidates.length === 0) return null;

    const minAbsDiff = Math.min(...candidates.map(c => Math.abs(2 * c - totalLen)));
    const bestCandidates = candidates.filter(c => Math.abs(2 * c - totalLen) === minAbsDiff);

    if (bestCandidates.length === 1) return bestCandidates[0];

    // 複数候補時は priority 適用
    return priority === 1 
      ? bestCandidates.find(c => 2 * c >= totalLen)! 
      : bestCandidates.find(c => 2 * c < totalLen)!;
  }

  /**
   * デフォルトの中間分割位置を取得する
   */
  private getDefaultSplitIndex(totalLen: number): number {
    return Math.ceil(totalLen / 2);
  }
```

### ご提示のループ処理に関する技術的評価

ご提示いただいた `splitter` によるループ処理は、論理的・技術的に**以下の2点を修正すれば実現可能**です。

1.  **インデックスの不一致（String vs Grapheme）**:
    `content.indexOf('｜')` は「文字数（UTF-16コードユニット数）」を返しますが、`graphemes.slice()` は「書記素数」を期待します。サロゲートペア等が含まれる場合、位置がズレます。
    *   **対策**: パイプの検索も `graphemes` 配列に対して行います。
2.  **メタ文字の処理（Skip）**:
    パイプ分割の場合は「パイプそのもの」を消す（1文字スキップする）必要がありますが、単語分割や中間分割は「文字の間」で分ける（0文字スキップする）必要があります。
    *   **対策**: 戻り値を `{ index: number, skip: number }` というオブジェクトにします。

```typescript
  /**
   * 割注のコンテンツを二行に分割する
   */
  private splitContent(content: string): [string, string] {
    const graphemes = this.getSegments(content, 'grapheme');
    const totalLen = graphemes.length;

    // 分割戦略の定義: [メソッド名, 引数リスト]
    const strategies: [keyof Warichu, any[]][] = [
      ['getPipeSplitResult', [graphemes]],
      ['getWordSplitResult', [content, totalLen]],
      ['getDefaultSplitResult', [totalLen]]
    ];

    for (const [method, args] of strategies) {
      // @ts-ignore: 内部メソッドへの動的アクセス
      const res = (this[method] as Function)(...args) as { index: number, skip: number } | null;
      
      if (res !== null) {
        const start = graphemes.slice(0, res.index).join('');
        const end = graphemes.slice(res.index + res.skip).join('');

   * パイプによる分割位置を取得（メタ文字を1文字スキップする）

    if (index === -1) return null;

      this.throwError('割注にパイプ｜が複数あります。一個まで有効です。');
    }

    return { index, skip: 1 };
  }

  /**

   */

    const { diff = 1, priority = 1 } = this._.options.split!;
    const wordSegments = this.getSegments(content, 'word');
    if (wordSegments.length <= 1) return null;

    let candidates: number[] = [];
    let currentLen = 0;
    for (let i = 0; i < wordSegments.length - 1; i++) {
      currentLen += this.getSegments(wordSegments[i], 'grapheme').length;
      const absDiff = Math.abs(2 * currentLen - totalLen);
      if (absDiff <= diff) candidates.push(currentLen);
    }

    if (candidates.length === 0) return null;

    const minAbsDiff = Math.min(...candidates.map(c => Math.abs(2 * c - totalLen)));
    const bestCandidates = candidates.filter(c => Math.abs(2 * c - totalLen) === minAbsDiff);

    const index = bestCandidates.length === 1 
      ? bestCandidates[0]
      : (priority === 1 
          ? bestCandidates.find(c => 2 * c >= totalLen)! 
          : bestCandidates.find(c => 2 * c < totalLen)!);

    return { index, skip: 0 };
  }

  /**
   * デフォルトの中間分割位置を取得（スキップなし）
   */
  private getDefaultSplitResult(totalLen: number): { index: number, skip: number } {
    return { index: Math.ceil(totalLen / 2), skip: 0 };
  }
```

ご提案いただいた「`skip` を戦略定義側に持たせる」という構造は、各メソッドの責務を「分割位置の特定」のみに純化させ、戻り値を `number | null` に統一できるため、非常に合理的です。

### 修正内容：`splitContent` とサブメソッドの再構築

ご指示通り、`skip` を `strategies` 配列に含め、各メソッドはインデックスのみを返す形に修正します。

```typescript
  /**
   * 割注のコンテンツを二行に分割する
   */
  private splitContent(content: string): [string, string] {
    const graphemes = this.getSegments(content, 'grapheme');
    const totalLen = graphemes.length;

    // 戦略定義: [メソッド名, 引数リスト, スキップ文字数]
    const strategies: [keyof Warichu, any[], number][] = [
      ['getPipeSplitIndex', [graphemes], 1],
      ['getWordSplitIndex', [content, totalLen], 0],
      ['getDefaultSplitIndex', [totalLen], 0]
    ];

    for (const [method, args, skip] of strategies) {
      // @ts-ignore: 内部メソッドへの動的アクセス
      const index = (this[method] as Function)(...args) as number | null;
      
      if (index !== null) {
        return [
          graphemes.slice(0, index).join(''),
          graphemes.slice(index + skip).join('')
        ];
      }
    }
    
    throw new WarichuError('分割処理に失敗しました。');
  }

  /**
   * パイプによる分割位置を取得
   */
  private getPipeSplitIndex(graphemes: string[]): number | null {
    const index = graphemes.indexOf('｜');
    if (index === -1) return null;

    // バリデーション: 複数は禁止
    if (graphemes.indexOf('｜', index + 1) !== -1) {
      this.throwError('割注にパイプ｜が複数あります。一個まで有効です。');
    }

    // バリデーション: 端にある（片方が空になる）のは禁止
    if (index === 0 || index === graphemes.length - 1) {
      const start = graphemes.slice(0, index).join('');
      const end = graphemes.slice(index + 1).join('');
      this.throwError(`割注の片方の行が空です。: start:${start}, end:${end}`);
    }

    return index;
  }

  /**
   * 単語境界による分割位置を取得
   */
  private getWordSplitIndex(content: string, totalLen: number): number | null {
    const { diff = 1, priority = 1 } = this._.options.split!;
    const wordSegments = this.getSegments(content, 'word');
    if (wordSegments.length <= 1) return null;

    let candidates: number[] = [];
    let currentLen = 0;
    for (let i = 0; i < wordSegments.length - 1; i++) {
      currentLen += this.getSegments(wordSegments[i], 'grapheme').length;
      const absDiff = Math.abs(2 * currentLen - totalLen);
      if (absDiff <= diff) candidates.push(currentLen);
    }

    if (candidates.length === 0) return null;

    const minAbsDiff = Math.min(...candidates.map(c => Math.abs(2 * c - totalLen)));
    const bestCandidates = candidates.filter(c => Math.abs(2 * c - totalLen) === minAbsDiff);

    return bestCandidates.length === 1 
      ? bestCandidates[0]
      : (priority === 1 
          ? bestCandidates.find(c => 2 * c >= totalLen)! 
          : bestCandidates.find(c => 2 * c < totalLen)!);
  }

  /**
   * デフォルトの中間分割位置を取得
   */
  private getDefaultSplitIndex(totalLen: number): number {
    return Math.ceil(totalLen / 2);
  }
```

### 修正のポイント
1.  **インターフェースの統一**: すべてのサブメソッドが `number | null`（デフォルトのみ `number`）を返すようになり、メソッド名 `Index` と実態が一致しました。
2.  **DRYな制御**: `skip` の値が `splitContent` のループ内で一括処理されるため、各サブメソッド内でオブジェクトを生成して返す手間が省け、コードが簡潔になりました。
3.  **型安全性の向上**: `skip` を戦略定義に持たせることで、どのメソッドがメタ文字を消費し、どのメソッドが消費しないかが一目で分かるようになりました。

これで、構造的な美しさと要件への忠実さが両立されたはずです。ご確認をお願いします。

### 到達不能なパスに対するエラーについて

ご指摘の通り、`getDefaultSplitIndex` が必ず数値を返す設計である以上、その後の `throw` は**「論理的に到達してはならないパス」**です。

将来の拡張（例えば、特定の条件下で全分割をスキップするロジックの追加など）でバグが混入し、どの戦略も `null` を返してしまった場合に、原因不明の挙動（`undefined` を返すなど）になるのを防ぐための「防波堤」としての役割になります。

### 論理エラーの名称案

「ユーザーの入力ミス」ではなく「プログラムの構造上の欠陥」であることを示す名前として、以下の案を提示します。

1.  **`UnreachableError`**
    *   「到達不能（Unreachable）なはずのコードに達した」という、コンパイラや型安全の文脈でよく使われる非常に明確な名前です。
2.  **`ImplementationError`**
    *   「実装（Implementation）の不備」であることを示します。ライブラリ利用者の責任ではなく、開発者の責任であることを強調できます。
3.  **`InconsistencyError`**
    *   「内部状態の不整合（Inconsistency）」を意味します。ロジックが矛盾していることを示します。
4.  **`WarichuLogicError`**（ご提示の `LogicError` に接頭辞をつけたもの）
    *   ライブラリ固有の論理エラーであることが分かりやすく、既存の `WarichuError` との親和性も高いです。

**私の推奨:**
最もこのエラーの性質（「ここに来るはずがない」）を正確に表しているのは **`UnreachableError`** だと考えます。
